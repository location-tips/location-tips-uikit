# Definition

Full catalog of Mobiquity UI Kit icons available for developers

## Usage

Just press on the Icon you need and valid code will be in your clipboard.

## Icons

{
    Object.keys(IconsCatalog).map((catalog, i) => {
        return <div class="catalog">
            <h3>{catalog}</h3>

            <div class="list-of-items">
                {
                    IconsCatalog[catalog].map((name) => {
                        return <div class="name">
                            <h4>{name}</h4>
                            <div>
                            {
                                Array.from(IconModeIndex.keys()).map((mode) => {
                                    return <div class="mode" onClick={() =>  navigator.clipboard.writeText(`<MIcon catalog="${catalog}" name="${name}" mode="${mode}" width={32} />`)}>
                                        <MIcon catalog={catalog} name={name} mode={mode} width={32} />
                                        <h5>{mode}</h5>
                                        <div class="copy">
                                            <MIcon catalog="Office & Editing" name="Copy" mode="regular" width={16} />
                                        </div>
                                    </div>
                                })
                            }
                            </div>
                        </div>
                    })
                }
            </div>
        </div>
    })
}
